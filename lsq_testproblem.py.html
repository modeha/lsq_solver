def dcttm(B):
"""
:rtype : object
:param B:
:return:
"""
B = as_llmat(B)
B = PysparseMatrix(matrix=B)
[M,N] = B.shape
A = np.zeros([M,N])
for m in xrange(0, M):
for n in xrange(0, N):
c = alpha(m,M)*alpha(n,N)
a = g(M,m)
b = g(N,n)
Bb = B*b.T[:,0]
A[m][n] = c*np.dot(a,Bb)#dot(a,B.T)#*b.T
return A
def alpha(p,n):
"""
:param p:
:param n:
:return:
"""
if p==0:
m = sqrt(1/float(n))
else:
m = sqrt(2/float(n))
return m
def g(M,p):
"""
:param M:
:param p:
:return:
"""
b = np.zeros([1,M])
for x in xrange(0, M):
b[0][x] = float(cos(p*pi * (2 * x + 1) / (2 * M)))
return b
